The root cause is a desync between CSS `transform` rendering and JS `getBoundingClientRect()` measurements, alongside CSS specificity overriding our Z-index logic. We are replacing the entire collision and positioning engine with a deterministic, absolute pixel-math approach that circumvents CSS completely.

        function resolveSmartCollisions() {
            const containers = Array.from(document.querySelectorAll('.marker-imp-container'));
            if (containers.length === 0) return;

            const trackRect = document.getElementById('track').getBoundingClientRect();
            const allMarkers = Array.from(document.querySelectorAll('.film-marker'));

            // Establish Sky Baseline
            let minMarkerY = trackRect.top;
            allMarkers.forEach(m => {
                const rect = m.getBoundingClientRect();
                if (rect.top < minMarkerY) minMarkerY = rect.top;
            });
            const skyBaseline = minMarkerY - 25;

            // 1. PRE-CALCULATE absolute X positions to decouple from CSS transform quirks
            const boxesData = containers.map(box => {
                const marker = box.parentElement;
                const label = box.querySelector('.marker-label');
                const svg = box.querySelector('.marker-pole-svg');

                // Temporarily wipe CSS transforms to measure raw element width safely
                label.style.transform = 'none';
                label.style.left = '0px';

                const markerRect = marker.getBoundingClientRect();
                const markerCenterX = markerRect.left + (markerRect.width / 2);
                const labelWidth = label.offsetWidth;

                // Center label mathematically over the dot
                let absoluteLeft = markerCenterX - (labelWidth / 2);
                let absoluteRight = absoluteLeft + labelWidth;

                // Apply Viewport Edge Bounding
                const safetyPadding = 20;
                const viewportWidth = window.innerWidth;

                if (absoluteLeft < safetyPadding) {
                    absoluteLeft = safetyPadding;
                    absoluteRight = absoluteLeft + labelWidth;
                } else if (absoluteRight > viewportWidth - safetyPadding) {
                    absoluteRight = viewportWidth - safetyPadding;
                    absoluteLeft = absoluteRight - labelWidth;
                }

                // Calculate the exact local pixel offset required
                const localLeft = absoluteLeft - markerCenterX;

                return { box, marker, label, svg, absoluteLeft, absoluteRight, localLeft, markerRect };
            });

            // 2. Sort by absolute X for a flawless collision sweep
            boxesData.sort((a, b) => a.absoluteLeft - b.absoluteLeft);

            const lanes = [];

            boxesData.forEach(data => {
                const { box, marker, label, svg, absoluteLeft, absoluteRight, localLeft, markerRect } = data;
                const gap = 15;

                // 3. Find Lane using true absolute bounds
                let laneIndex = 0;
                let found = false;
                while (!found) {
                    if (!lanes[laneIndex]) lanes[laneIndex] = [];
                    const hasCollision = lanes[laneIndex].some(zone => {
                        return !(absoluteRight + gap < zone.left || absoluteLeft - gap > zone.right);
                    });
                    if (!hasCollision) found = true;
                    else laneIndex++;
                }
                lanes[laneIndex].push({ left: absoluteLeft, right: absoluteRight });

                // 4. Base Anchoring
                box.style.top = '50%';
                box.style.left = '50%';
                box.style.bottom = 'auto';
                box.style.transform = 'none'; 

                // 5. CRITICAL Z-INDEX FIX: Force stacking context with !important
                marker.style.setProperty('z-index', 1000 - (laneIndex * 10), 'important');
                box.style.setProperty('z-index', 'auto', 'important');
                label.style.setProperty('z-index', '10', 'important');

                // 6. Height Calculation
                const distanceToSky = markerRect.top - skyBaseline;
                const step = 42;
                const safeDistance = Math.max(20, distanceToSky);
                const labelHeightOffset = safeDistance + (laneIndex * step) + 6;

                // 7. Apply Deterministic CSS to Label
                label.style.bottom = `${labelHeightOffset}px`;
                label.style.left = `${localLeft}px`;
                label.style.transform = 'none'; // Position is now driven entirely by exact absolute left

                // 8. SVG Path Generation
                const startY = -labelHeightOffset + 2;
                const dropX = -14;
                const dotCenterY = 0;
                const dotEdgeX = -6;

                svg.innerHTML = `
                    <path d="M ${dropX} ${startY} L ${dropX} ${dotCenterY} L ${dotEdgeX} ${dotCenterY}"
                          fill="none"
                          stroke="rgba(255, 255, 255, 0.4)"
                          stroke-width="1.5" />
                `;
            });
        }
