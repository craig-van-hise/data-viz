Additionally, the collision engine is failing because it uses fixed base pole heights. Since the dots start at different vertical heights within the waveform, the poles must dynamically calculate their length to reach a uniform "Sky Baseline" so labels align perfectly horizontally.

### 2. Implement the "Sky Baseline" Collision Logic (JS)
Locate the `resolveSmartCollisions` function. We are going to replace everything inside this function to establish an absolute Y-coordinate baseline that all poles reach toward.

**Replace the entire `resolveSmartCollisions` function with the following:**

```javascript
        function resolveSmartCollisions() {
            const containers = Array.from(document.querySelectorAll('.marker-imp-container'));
            if (containers.length === 0) return;

            // 1. Find the highest peak of the waveform to establish our "Sky Baseline"
            const trackRect = document.getElementById('track').getBoundingClientRect();
            const allMarkers = Array.from(document.querySelectorAll('.film-marker'));
            
            let minMarkerY = trackRect.top;
            allMarkers.forEach(m => {
                const rect = m.getBoundingClientRect();
                if (rect.top < minMarkerY) minMarkerY = rect.top;
            });

            // The Sky Baseline is a safe distance (25px) above the highest dot
            const skyBaseline = minMarkerY - 25;

            // 2. Sort containers by horizontal position
            containers.sort((a, b) => {
                return a.getBoundingClientRect().left - b.getBoundingClientRect().left;
            });

            const lanes = [];

            containers.forEach(box => {
                const label = box.querySelector('.marker-label');
                const pole = box.querySelector('.marker-pole');
                const marker = box.parentElement; // The dot element this is attached to

                const rect = label.getBoundingClientRect();
                const width = rect.width;
                const left = rect.left;
                const right = left + width;
                const gap = 15; // Gap between labels

                // 3. Find available lane
                let laneIndex = 0;
                let found = false;

                while (!found) {
                    if (!lanes[laneIndex]) lanes[laneIndex] = [];

                    const hasCollision = lanes[laneIndex].some(zone => {
                        return !(right + gap < zone.left || left > zone.right + gap);
                    });

                    if (!hasCollision) found = true;
                    else laneIndex++;
                }
                lanes[laneIndex].push({ left, right });

                // 4. Dynamic Pole Height Calculation
                box.style.bottom = '15px'; // Anchor securely to the dot
                box.style.top = 'auto';

                // Calculate exact distance from this specific dot to the Sky Baseline
                const markerRect = marker.getBoundingClientRect();
                const distanceToSky = markerRect.top - skyBaseline;

                // Base pole reaches the baseline; laneIndex adds steps above the baseline
                const step = 42; // Height of each lane
                const safeDistance = Math.max(20, distanceToSky); // Fallback to prevent negative heights
                const newPoleHeight = safeDistance + (laneIndex * step);
                
                pole.style.height = `${newPoleHeight}px`;
                box.style.zIndex = 100 + laneIndex;

                // 5. Horizontal Edge Bounding
                const viewportWidth = window.innerWidth;
                const safetyPadding = 20;
                let currentTransform = -50;
                let pixelOffset = 0;

                if (left < safetyPadding) {
                    pixelOffset = safetyPadding - left;
                } else if (right > (viewportWidth - safetyPadding)) {
                    pixelOffset = (viewportWidth - safetyPadding) - right;
                }

                if (pixelOffset !== 0) {
                    box.style.transform = `translateX(calc(${currentTransform}% + ${pixelOffset}px))`;
                } else {
                    box.style.transform = `translateX(${currentTransform}%)`;
                }
            });
        }
```
