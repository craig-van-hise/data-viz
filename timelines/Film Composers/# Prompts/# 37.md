**Objective:**

2. **Inject HTML:** Add `<input type="text" id="globalSearch" class="search-box" placeholder="Search composer or film...">` inside the `<header>`.
3. **Data Hook:** In `createRows()`, where `const row = document.createElement('div');` is created, add: `row.setAttribute('data-composer', name.toLowerCase());`
4. **Logic:** Add this event listener at the bottom of the `<script>` tag:
   ```javascript
   document.getElementById('globalSearch').addEventListener('input', (e) => {
       const term = e.target.value.toLowerCase().trim();
       const rows = document.querySelectorAll('.composer-row');
       
       rows.forEach(row => {
           const composerName = row.getAttribute('data-composer');
           const myFilms = filmData.find(f => f.composer.toLowerCase() === composerName);
           
           let composerHit = composerName.includes(term);
           let filmHit = false;
           
           if (myFilms && myFilms.films) {
               filmHit = myFilms.films.some(f => f.title.toLowerCase().includes(term));
           }
           
           row.classList.remove('search-film-hit');
           
           if (term === '') {
               row.style.display = 'flex';
           } else if (composerHit) {
               row.style.display = 'flex';
           } else if (filmHit) {
               row.style.display = 'flex';
               row.classList.add('search-film-hit');
           } else {
               row.style.display = 'none';
           }
       });
   });
   ```

   ```javascript
   document.getElementById('profileSearch').addEventListener('input', (e) => {
       const term = e.target.value.toLowerCase().trim();
       const markers = document.querySelectorAll('.film-marker');
       
       markers.forEach(marker => {
           const title = marker.getAttribute('data-title');
           marker.classList.remove('search-dim', 'search-target');
           
           if (term !== '') {
               if (title.includes(term)) {
                   marker.classList.add('search-target');
               } else {
                   marker.classList.add('search-dim');
               }
           }
       });
   });
   ```
